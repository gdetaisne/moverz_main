# ==========================================
# MOVERZ - Configuration GA4 & BigQuery
# ==========================================
# Copier ce fichier vers .env et remplir avec vos vraies valeurs

# ==========================================
# 1. Google Analytics 4 API
# ==========================================
# Format: properties/123456789
GA4_PROPERTY_ID=properties/YOUR_PROPERTY_ID

# ==========================================
# 2. BigQuery (si vous utilisez l'export GA4)
# ==========================================
# ID du projet Google Cloud
BIGQUERY_PROJECT_ID=your-project-id

# ID du dataset GA4 (format: analytics_XXXXXXXXX)
BIGQUERY_DATASET_ID=analytics_XXXXXXXXX

# ==========================================
# 3. Authentification Google Cloud
# ==========================================
# Chemin vers le fichier JSON du service account
GOOGLE_APPLICATION_CREDENTIALS=./service-account-key.json

# ==========================================
# 4. Configuration du monitoring
# ==========================================
# Seuil minimum de vues pour créer une redirection
THRESHOLD_VIEWS=10

# Période d'analyse en jours
ANALYSIS_PERIOD_DAYS=7

# ==========================================
# INSTRUCTIONS:
# ==========================================
# 1. Créer un service account dans Google Cloud Console
#    https://console.cloud.google.com/iam-admin/serviceaccounts
# 
# 2. Activer les APIs:
#    - Google Analytics Data API (v1)
#    - BigQuery API
# 
# 3. Donner les permissions au service account:
#    - GA4: Viewer ou Analyst
#    - BigQuery: BigQuery Data Viewer
# 
# 4. Télécharger la clé JSON du service account
# 
# 5. Copier ce fichier:
#    cp env.example.txt .env
# 
# 6. Remplir les valeurs dans .env
